    function spawn(genF) {
  return new Promise(function(resolve, reject) {
    const gen = genF();
    function step(nextF) {
      let next;
      try {
        next = nextF();

      if (next.done) {
        return resolve(next.value);

        function(v) {

            return gen.next(v);
